11.  create userprofile function

public function userprofile()
{
  $user=Auth::user();
  if($user)
  {
   return response()->json([
            'status'=>'success',
            'message'=>'Profile data access successful  ',
            'data'=>$user
         ],200);
  }
  
  return response()->json([
                    'status'=>'Error',
                    'message'=>'Unauthenticated'
                  ],400);
}

12. logout function  

 public function logout()
{
  $user=Auth::user();
  if($user)
  {
   $user->currentAccessToken()->delete();
   return response()->json([
                        'status'=>'success',
                        'message'=>'Logout success'
                     ],200);
  }
  
  return response()->json([
                       'status'=>'Error',
                       'message'=>'Can not logout'
                     ],400);

}
